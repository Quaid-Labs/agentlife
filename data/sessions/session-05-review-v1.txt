  // ... existing CRUD endpoints stay the same ...

  // Filter by dietary tags
  app.get('/api/recipes/filter', (req, res) => {
    const tags = req.query.tags ? req.query.tags.split(',') : [];
    if (tags.length === 0) {
      const recipes = db.prepare('SELECT * FROM recipes ORDER BY created_at DESC').all();
      return res.json(recipes);
    }
    const recipes = db.prepare('SELECT * FROM recipes ORDER BY created_at DESC').all();
    const filtered = recipes.filter(r => {
      const recipeTags = r.dietary_tags ? r.dietary_tags.split(',') : [];
      return tags.every(t => recipeTags.includes(t));
    });
    res.json(filtered);
  });

  app.get('/api/recipes/safe-for-mom', (req, res) => {
    const { SAFE_FOR_MOM } = require('./database');

    const filtered = recipes.filter(r => {
      const recipeTags = r.dietary_tags ? r.dietary_tags.split(',') : [];
