AI ASSISTANT:
  Here's `migrate.js`:

  ```javascript
  const db = require('./database');
  const fs = require('fs');
  const path = require('path');

  const MIGRATIONS_DIR = path.join(__dirname, 'migrations');

  db.exec(`
    CREATE TABLE IF NOT EXISTS migrations (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      name TEXT NOT NULL UNIQUE,
      applied_at DATETIME DEFAULT CURRENT_TIMESTAMP
    )
  `);

  function getAppliedMigrations() {
    return db.prepare('SELECT name FROM migrations ORDER BY id')
      .all().map(r => r.name);
  }

  function getPendingMigrations() {
    const applied = new Set(getAppliedMigrations());
    const files = fs.readdirSync(MIGRATIONS_DIR)
      .filter(f => f.endsWith('.sql'))
      .sort();
    return files.filter(f => !applied.has(f));
  }

  function runMigrations() {
    const pending = getPendingMigrations();
    if (pending.length === 0) {
      console.log('No pending migrations.');
      return;
    }

    for (const file of pending) {
      const sql = fs.readFileSync(
        path.join(MIGRATIONS_DIR, file), 'utf8'
      );
      console.log(`Running migration: ${file}`);
      db.exec(sql);
      db.prepare('INSERT INTO migrations (name) VALUES (?)')
        .run(file);
      console.log(`  Applied: ${file}`);
    }

    console.log(`${pending.length} migration(s) applied.`);
  }

  if (require.main === module) { runMigrations(); }

  module.exports = { runMigrations, getPendingMigrations };
  ```

  Run it standalone with `node migrate.js` or it gets called on
  server startup. Migrations are idempotent — it only runs new ones.

ANALYSIS: Simple migration runner matching Maya's requirements. Tracks
applied migrations, runs pending ones in order. Added to server.js
startup via `require('./migrate')`.
