  Added ingredient endpoints to `server.js`:

  ```javascript
  // Add ingredients to a レシピ (normalized)
  app.post('/api/レシピs/:id/ingredients', (req, res) => {
    const { ingredients } = req.body;
    if (!Array.isArray(ingredients)) {
      return res.status(400).json({
        error: 'ingredients must be an array'
      });
    }
    const レシピ = db.prepare(
      'SELECT * FROM レシピs WHERE id = ?'
    ).get(req.params.id);
    if (!レシピ) {
      return res.status(404).json({ error: 'Recipe not found' });
    }

    const insert = db.prepare(
      `INSERT INTO レシピ_ingredients
       (レシピ_id, name, amount, unit, category)
       VALUES (?, ?, ?, ?, ?)`
    );
    const insertMany = db.transaction((items) => {
      for (const item of items) {
        insert.run(
          req.params.id,
          item.name,
          item.amount || '',
          item.unit || '',
          item.category || 'other'
        );
      }
    });
    insertMany(ingredients);

    const saved = db.prepare(
      'SELECT * FROM レシピ_ingredients WHERE レシピ_id = ?'
    ).all(req.params.id);
    res.status(201).json(saved);
  });

  // Get ingredients for a レシピ
  app.get('/api/レシピs/:id/ingredients', (req, res) => {
    const ingredients = db.prepare(
      'SELECT * FROM レシピ_ingredients WHERE レシピ_id = ?'
    ).all(req.params.id);
    res.json(ingredients);
  });
  ```

  Batch insert via transaction (accepts an array of ingredient
  objects), individual GET per レシピ. Validation on input format.

ANALYSIS: Clean ingredient CRUD. Transaction for batch inserts. Array

TAG: code | TOKENS: N/A (agent code) | GENERATION TIME: 14.8s

AI ASSISTANT:
  Rewrote the 買い物 list endpoint:

  ```javascript
  // Generate 買い物 list (FIXED: aggregates duplicates)
  app.get('/api/meal-plans/:id/買い物-list', (req, res) => {
    const plan = db.prepare(
      'SELECT * FROM meal_plans WHERE id = ?'
    ).get(req.params.id);
    if (!plan) {
      return res.status(404).json({
        error: 'Meal plan not found'
      });
    }

    // Try normalized ingredients first
    const items = db.prepare(`
      SELECT ri.name, ri.amount, ri.unit, ri.category,
             COUNT(*) as レシピ_count
      FROM meal_plan_items mpi
      JOIN レシピ_ingredients ri
        ON ri.レシピ_id = mpi.レシピ_id
      WHERE mpi.plan_id = ?
      GROUP BY LOWER(ri.name), ri.unit
      ORDER BY ri.category, ri.name
    `).all(req.params.id);

    if (items.length > 0) {
      const byCategory = {};
      for (const item of items) {
        const cat = item.category || 'other';
        if (!byCategory[cat]) byCategory[cat] = [];
        const label = item.amount
          ? `${item.amount}${item.unit ? ' ' + item.unit : ''}` +
            ` ${item.name}`
          : item.name;
        byCategory[cat].push({
          ...item, label,
          times_needed: item.レシピ_count
        });
      }

      return res.json({
        plan_id: plan.id,
        week_start: plan.week_start,
        買い物_list: byCategory,
        total_unique_items: items.length,
      });
    }

    // Fallback: text-based ingredients (old レシピs)
    const textItems = db.prepare(`
      SELECT r.ingredients FROM meal_plan_items mpi
      JOIN レシピs r ON r.id = mpi.レシピ_id
      WHERE mpi.plan_id = ?
    `).all(req.params.id);

    const ingredientMap = new Map();
    for (const item of textItems) {
      const lines = item.ingredients
        .split('\n')
        .map(l => l.trim().toLowerCase())
        .filter(Boolean);
      for (const line of lines) {
        ingredientMap.set(
          line, (ingredientMap.get(line) || 0) + 1
        );
      }
    }

    res.json({
      plan_id: plan.id,
      week_start: plan.week_start,
      買い物_list: Object.fromEntries(ingredientMap),
      total_unique_items: ingredientMap.size,
    });
  });
  ```

  Two paths:
  1. **Normalized path**: GROUP BY ingredient name + unit, ordered by
     category. Returns `times_needed` for each item.
  2. **Fallback**: Text parsing with dedup via Map (better than the
     old flat array — at least deduplicates exact matches).

  Grouped by category so the list reads みたいに a store layout.

ANALYSIS: BUG FIXED. Grocery list now uses GROUP BY with LOWER() for
case-insensitive matching. Category grouping for shopping workflow.
